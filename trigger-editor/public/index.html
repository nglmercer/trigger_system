🎯 Trigger Editor

A browser-based visual editor for Trigger System rules

Quick API Examples:

// Get all rules
const rules = editor.getRules();

// Add a rule programmatically
editor.addRule({
  id: 'welcome-rule',
  name: 'Welcome Message',
  on: 'user.signup',
  enabled: true,
  priority: 10,
  do: { type: 'notify', params: { message: 'Welcome!' } }
});

// Export as YAML
const yaml = editor.exportYaml();
console.log(yaml);

// Listen for events
editor.addEventListener('rule-added', (e) => {
  console.log('Rule added:', e.detail);
});

editor.addEventListener('rules-changed', (e) => {
  console.log('Rules updated:', e.detail);
});
